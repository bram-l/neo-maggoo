    lib/utils/extendClass.js

    'use strict'

const CLASS_PROPERTIES = ['length', 'name', 'prototype']
const PROTOTYPE_PROPERTIES = ['constructor']

function extend(Target, Source, excluded)
{
	const keys = Object.getOwnPropertyNames(Source)

	for (const key of keys)
	{
		if (excluded.includes(key))
		{
			continue
		}

		const descriptor = Object.getOwnPropertyDescriptor(Source, key)

		Object.defineProperty(Target, key, descriptor)
	}
}

module.exports = (Target, Source) =>
{
	extend(Target, Source, CLASS_PROPERTIES)
	extend(Target.prototype, Source.prototype, PROTOTYPE_PROPERTIES)
}
